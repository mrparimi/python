## Adding spaces
```
def clean_sentence(sent):
    text = str(sent)
    text = text.replace('/', " / ")
    text = text.replace('-', ' - ')
    text = text.replace('&', ' & ')
    text = text.replace('?', " ?")
    text = text.replace(',', " , ")
    text = text.replace('\t', " ")
    text = text.replace('\n', " ")
    text = text.replace('$', " $ ")
    return text
```
